새로운 태그생성

확인 버튼 클릭하면 p태그를 생성 result1에 추가
ptag = $('<p id="p1" onclick="proc1(this)">Hello</p>');
$('#result1').append(ptag);

function proc1(tt){
$(tt).css('font-size', '+=5px');
}
